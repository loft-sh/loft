import { Step } from '@site/src/components/Flow'
import NavStep from '@site/src/components/NavStep'
import Button from '@site/src/components/Button'
import Label from '@site/src/components/Label'

<Step>
    Select the <NavStep>Projects</NavStep> field on the left menu bar.
</Step>
<Step>
    Select the project containing the space that you'd like to install
    the App in from the Project drop down menu.
</Step>
<Step>
    From the Projects sub-menu, select the Spaces option.
</Step>
<Step>
    Find the space you would like to deploy an App to in the list of spaces, and click on that
    space.
</Step>
<Step>
    In the Space Management view, click the <Button>Apps</Button> button.
</Step>
<Step>
    Select the App you would like to install from the App thumbnails, or, if there
    are no recommend Apps, click the <Button>Install App</Button> and find  the
    desired App in the <Label>Select an App</Label> drop down menu.
</Step>
<Step>
    Enter any other required parameters for your App -- this may vary depending on
    the type of App and whether or not the App creator has specified parameters.
</Step>
<Step>
    Click the <Button>Install</Button> button.
</Step>
